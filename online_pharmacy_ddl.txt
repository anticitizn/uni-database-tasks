DROP DATABASE IF EXISTS pharmacy;

CREATE DATABASE IF NOT EXISTS pharmacy CHARACTER SET UTF8mb4 COLLATE utf8mb4_bin;
USE pharmacy;

CREATE TABLE boxes_storage (
	box_id VARCHAR(3) NOT NULL,
	storage_id VARCHAR(3) NOT NULL,
	PRIMARY KEY (box_id)
) ENGINE=InnoDB;

CREATE TABLE drugs (
	drug_number INT(6) NOT NULL,
	name VARCHAR(255) NOT NULL,
	price FLOAT(4) NOT NULL,
	box_id VARCHAR(3) NOT NULL,
	quantity INT(10),
	PRIMARY KEY (drug_number)
) ENGINE=InnoDB;

ALTER TABLE drugs ADD CONSTRAINT fk_boxes_storage_id
FOREIGN KEY (box_id)
REFERENCES boxes_storage(box_id)
ON UPDATE CASCADE
ON DELETE CASCADE;

CREATE TABLE cities_zip (
	zip_code INT(5) NOT NULL,
	city VARCHAR(255) NOT NULL,
	PRIMARY KEY (zip_code)
) ENGINE=InnoDB;

CREATE TABLE customers (
	id INT(10) NOT NULL,
	name VARCHAR(255) NOT NULL,
	street VARCHAR(255) NOT NULL,
	zip_code INTEGER(5) NOT NULL,
	date_of_birth DATE NOT NULL,
	service_pin INT(4) NOT NULL,
	PRIMARY KEY(id)
) ENGINE=InnoDB;

ALTER TABLE customers ADD CONSTRAINT fk_customers_zip
FOREIGN KEY (zip_code)
REFERENCES cities_zip(zip_code)
ON UPDATE CASCADE
ON DELETE CASCADE;

CREATE TABLE orders (
	id INT(10) NOT NULL,
	customer_id INT(10) NOT NULL,
	order_date DATE NOT NULL,
	PRIMARY KEY (id)
) ENGINE=InnoDB;

ALTER TABLE orders ADD CONSTRAINT fk_orders_customer_id
FOREIGN KEY (customer_id)
REFERENCES customers(id)
ON UPDATE CASCADE
ON DELETE CASCADE;

CREATE TABLE orders_drugs (
	id INT(10) NOT NULL,
	drug_id INT(10) NOT NULL,
	quantity INT(10) NOT NULL,
	PRIMARY KEY (id, drug_id)
) ENGINE=InnoDB;

ALTER TABLE orders_drugs ADD CONSTRAINT fk_orders_id
FOREIGN KEY (id)
REFERENCES orders(id)
ON UPDATE CASCADE
ON DELETE CASCADE;

ALTER TABLE orders_drugs ADD CONSTRAINT fk_order_drug_id
FOREIGN KEY (drug_id)
REFERENCES drugs(drug_number)
ON UPDATE CASCADE
ON DELETE CASCADE;
